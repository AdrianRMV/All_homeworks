//Preguntar por perimetro de figuras!

    int opcion;

    cout<<"Cual de las siguientes opciones desea realizar: ";
    cout<<"1.- perimetro de un cuadrado  2.-perimetro de un Rectangulo   3.- perimetro de un circulo  4.- perimetro de un triangulo"; cin>>opcion;

    switch ( opcion ){
        case 1:

            cout<<"Tu opcion fue: perimetro de un cuadrado";
            int lado,area;
            cout<<"Tu opcion fue: Area de un cuadrado";
            cout<<"Por favor dijite un lado de su cuadrado: "; cin>>lado;
            area = lado*lado;
            cout<<"El area de su cuadrado es: "<<area<<endl;
            break;


        case 2:
            cout<<"Tu opcion fue: perimetro de un rectangulo";
            int b,h,p2;
            cout<<"Cual es la base de su rectangulo: "; cin>>b;
            cout<<"Cual es la altura de su rectangulo: "; cin>>h;

            p2 = 2*(b+h);

            cout<<"El perimetro de su rectangulo es: "<<p2<<endl;
            break;

        case 3:

            cout<<"Tu opcion fue: perimetro de un circulo";
            int r,p3;
            cout<<"Cual es el radio de su circulo: "; cin>>r;

            p3 = 2*3.1416*r;

            cout<<"El perimetro de su circulo es: "<<p3<<endl;
            break;

        case 4:
            cout<<"Tu opcion fue: perimetro de un triangulo";

            int l1,l2,l3,p4;

            cout<<"Dijite el primer lado de su triangulo: "; cin>>l1;
            cout<<"Dijite el segundo lado de su triangulo: "; cin>>l2;
            cout<<"Dijite el tercer lado de su triangulo: "; cin>>l3;

            p4 = l1+l2+l3;

            cout<<"El perimetro de su triangulo es: "<<p4<<endl;

            break;







    }
}
