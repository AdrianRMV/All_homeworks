//Area del terreno

float a,b,c,area,a1,b1,c1,s;

cout<<"Ingrese el valor de A: "; cin>>a;
cout<<"Ingrese el valor de B: "; cin>>b;
cout<<"Ingrese el valor de C: "; cin>>c;


//Sacando La hipotenusa de la forma B para tener los 3 lados
a-=c;
a1 = a*a;
b1= (b*b);
c1= sqrt(a1+b1);

//Formula de heron

s = (a+b+c1)/2;

area = sqrt(s*(s-a)*(s-b)*(s-c1));

cout<<"El area de la Forma B es: "<<area<<endl;
