    char vocales[10];
    int numeroV=0,total=0;
    for (int i = 0; i < 10; ++i) {
        cout<<"Ingrese letra: "; cin>>vocales[i];
    }
    for (int i = 0; i < 10; ++i) {
        if (vocales[i] == 'a' or vocales[i] == 'e' or vocales[i] == 'i' or vocales[i] == 'o' or vocales[i] == 'u'){
            numeroV += 1;
        }
        total=numeroV;


    }
    cout<<"Total de vocales: "<<total;
