	
int target = 0;
int iteraciones = 0,x=0;

while (x < 100){
    iteraciones = 0;
    target = x;
    for (int i = 1; i <= target; ++i) {
        if(target % i == 0 ){
            iteraciones++;
        }
    }

    if(iteraciones == 2){
        cout<<"El numero"<<target<<"es primo"<<endl;

    }
    x++;
}
